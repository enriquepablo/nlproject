

    >>> from nl import Thing, Proposition, Rule, kb
    
    >>> from nl.docs.wines import *

    >>> CentralCoastalRegion = Region('CentralCoastalRegion')

    >>> CabernetSauvignonGrape = WineGrape('CabernetSauvignonGrape')

    >>> SantaCruzMountainsRegion = Region('SantaCruzMountainsRegion')
    >>> CaliforniaRegion = Region('CaliforniaRegion')

    >>> p1 = Proposition(SantaCruzMountainsRegion,
    ...                  LocatedIn(region=CaliforniaRegion),
    ...                  0)

    >>> SantaCruzMountainVineyard = Winery('SantaCruzMountainVineyard')

    >>> SantaCruzMountainVineyardCabernetSauvignon = Wine('SantaCruzMountainVineyardCabernetSauvignon')

    >>> p2 = Proposition(SantaCruzMountainVineyardCabernetSauvignon,
    ...                  LocatedIn(region=SantaCruzMountainsRegion),
    ...                  0)

    >>> p3 = Proposition(SantaCruzMountainVineyardCabernetSauvignon,
    ...                  HasMaker(maker=SantaCruzMountainVineyard),
    ...                  0)

    >>> madrid = Region('madrid')
    >>> sevilla = Region('sevilla')

    >>> r1 = Rule('rr1',
    ...           [Proposition(Thing('X1'), LocatedIn(region=sevilla), 0)],
    ...           [Proposition(Thing('X1'), LocatedIn(region=madrid), 0, truth=0)])



    >>> r2 = Rule('rr2',
    ...           [Proposition(Thing('X1'), LocatedIn(region=sevilla), 0, truth=0)],
    ...           [Proposition(Thing('X1'), LocatedIn(region=madrid), 0)])

    >>> cibeles = Thing('cibeles')
    >>> giralda = Thing('giralda')

    >>> pp1 = Proposition(giralda, LocatedIn(region=sevilla), 0)
    >>> pp2 = Proposition(cibeles, LocatedIn(region=sevilla), 0, truth=0)

    >>> kb.open()
    >>> kb.tell(madrid, sevilla, r1, r2, cibeles, giralda, pp1, pp2)
    >>> kb.extend()
    ...

    >>> kb.ask(Proposition(cibeles, LocatedIn(region=madrid), 0))
    'cibeles locatedin region madrid at 0.0'

    >>> kb.ask(Proposition(giralda, LocatedIn(region=madrid), 0, truth=0))
    'giralda not locatedin region madrid at 0.0'

    >>> kb.ask(Proposition(giralda, LocatedIn(region=madrid), 0))
    'no'

    >>> kb.close()
    'DB closed'
